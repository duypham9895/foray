### Task 1: Update Prisma schema

```prisma
model Document {
  id String @id @default(cuid())
  applicationId String
  application Application @relation(fields: [applicationId], references: [id], onDelete: Cascade)
  kind String // "resume", "cover_letter", "jd", "submission"
  filename String
  mimeType String
  fileSize Int
  version Int @default(1)
  uploadedAt DateTime @default(now())
  uploadedBy String // userId
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  
  @@index([applicationId])
  @@unique([applicationId, kind, version])
}
```

### Task 2: Create file storage helper

```typescript
// src/core/storage/documents.ts
export async function saveDocument(
  userId: string,
  applicationId: string,
  file: File,
  kind: string
) {
  const dir = `data/uploads/${userId}/${applicationId}`;
  const filename = `${kind}-v${version}.${ext}`;
  
  await fs.promises.mkdir(dir, { recursive: true });
  await fs.promises.writeFile(`${dir}/${filename}`, await file.arrayBuffer());
  
  return await db.document.create({
    data: {
      applicationId,
      kind,
      filename,
      mimeType: file.type,
      fileSize: file.size,
      uploadedBy: userId,
    },
  });
}
```

### Task 3: Add storage quota check

```typescript
export async function getStorageQuota(userId: string) {
  const total = await db.document.aggregate({
    where: { uploadedBy: userId },
    _sum: { fileSize: true },
  });
  
  return {
    usedBytes: total._sum.fileSize || 0,
    limitBytes: 1_000_000_000, // 1GB
  };
}
```
