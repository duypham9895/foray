**Step 3: Create src/features/documents/schema.ts**

```typescript
import { z } from 'zod'

export const MAX_FILE_SIZE = 10 * 1024 * 1024 // 10MB

export const documentKindEnum = z.enum([
  'resume', 'cover_letter', 'jd_pdf', 'take_home', 'other',
])

export type DocumentKind = z.infer<typeof documentKindEnum>

// Magic-byte MIME allowlist. Checked against first 4-8 bytes of file, NOT extension.
export const ALLOWED_MIME_TYPES: Record<string, { ext: string; magic: number[] }> = {
  'application/pdf':                                                          { ext: 'pdf', magic: [0x25, 0x50, 0x44, 0x46] },
  'application/msword':                                                       { ext: 'doc', magic: [0xD0, 0xCF, 0x11, 0xE0] },
  'application/vnd.openxmlformats-officedocument.wordprocessingml.document': { ext: 'docx', magic: [0x50, 0x4B, 0x03, 0x04] },
  'text/plain':                                                               { ext: 'txt', magic: [] },  // fallback: checked last
  'image/png':                                                                { ext: 'png', magic: [0x89, 0x50, 0x4E, 0x47] },
  'image/jpeg':                                                               { ext: 'jpg', magic: [0xFF, 0xD8, 0xFF] },
}

export const uploadSchema = z.object({
  kind: documentKindEnum,
  notes: z.string().max(2000).optional().or(z.literal('')),
})
```

**Step 4: Create src/features/documents/service.ts**

Implement four functions, all returning `Result<T, AppError>`:

1. `detectMimeType(buffer: ArrayBuffer): string | null`
   - Read first 8 bytes, check against ALLOWED_MIME_TYPES magic signatures in order
   - PDF: `[0x25, 0x50, 0x44, 0x46]` at offset 0
   - PNG: `[0x89, 0x50, 0x4E, 0x47]` at offset 0
   - JPEG: `[0xFF, 0xD8, 0xFF]` at offset 0
   - DOCX/ZIP: `[0x50, 0x4B, 0x03, 0x04]` at offset 0
   - DOC/OLE2: `[0xD0, 0xCF, 0x11, 0xE0]` at offset 0
   - Plain text: check if all bytes are printable ASCII (0x20-0x7E) or whitespace (0x09, 0x0A, 0x0D) — LAST resort
   - Return null if no match

2. `sanitizeFilename(name: string): string`
   - Replace `[^a-zA-Z0-9._-]` with `_`
   - Limit to 200 chars
   - Reject empty result

3. `uploadDocument(userId: UserId, applicationId: ApplicationId, file: File, kind: DocumentKind, notes?: string): Promise<Result<{ documentId: number; eventId: number }, AppError>>`
   - Validate file size: `if (file.size > MAX_FILE_SIZE) return err(errors.validation([{ code: 'custom', message: 'File exceeds 10MB limit', path: ['file'] }]))`
   - Read first 8 bytes: `const header = await file.slice(0, 8).arrayBuffer()`
   - Validate MIME: `const detectedMime = detectMimeType(header)` — if null or not in ALLOWED_MIME_TYPES, return err(validation)
   - Use `withRls(userId, async (tx) => { ... })`:
     a. Verify application belongs to user: `const app = await tx.application.findUnique({ where: { id: Number(applicationId) } })` — if !app or app.userId !== Number(userId), return err(notFound)
     b. Create Document row (without storagePath initially): `const doc = await tx.document.create({ data: { applicationId: Number(applicationId), kind, filename: sanitizeFilename(file.name), mimeType: detectedMime, storagePath: '', sizeBytes: file.size, notes: notes || null } })`
     c. Build storage path: `path.join(process.cwd(), 'data', 'documents', String(applicationId), String(doc.id), sanitizeFilename(file.name))`
     d. Ensure directory: `await fs.mkdir(path.dirname(storagePath), { recursive: true })`
     e. Write file: `await fs.writeFile(storagePath, Buffer.from(await file.arrayBuffer()))`
     f. Update Document with storagePath: `await tx.document.update({ where: { id: doc.id }, data: { storagePath } })`
     g. Create Event: `const event = await tx.event.create({ data: { applicationId: Number(applicationId), type: 'document_uploaded', data: { documentId: doc.id, filename: doc.filename, kind }, source: 'manual', occurredAt: new Date() } })`
     h. Return { documentId: doc.id, eventId: event.id }
   - On error after file write, attempt cleanup: `await fs.rm(storagePath, { force: true }).catch(() => {})`

4. `deleteDocument(userId: UserId, documentId: number): Promise<Result<void, AppError>>`
   - Use `withRls(userId, async (tx) => { ... })`:
     a. Find document with application: `const doc = await tx.document.findUnique({ where: { id: documentId }, include: { application: { select: { userId: true } } } })`
     b. If !doc or doc.application.userId !== Number(userId), return err(notFound)
     c. Delete Document row: `await tx.document.delete({ where: { id: documentId } })`
     d. Delete file from disk: `await fs.rm(doc.storagePath, { force: true })`
   - Return ok(undefined)

5. `listDocuments(userId: UserId, applicationId: ApplicationId): Promise<Result<Document[], AppError>>`
   - Use `withRls(userId, async (tx) => { ... })`:
     a. Verify application belongs to user
     b. Return `tx.document.findMany({ where: { applicationId: Number(applicationId) }, orderBy: { createdAt: 'desc' } })`

6. `getDocument(userId: UserId, documentId: number): Promise<Result<Document, AppError>>`
   - Use `withRls(userId, async (tx) => { ... })`:
     a. Find document with application userId check (same as deleteDocument)
     b. Return document or err(notFound)
