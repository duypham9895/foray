### Task 1: Create DocumentUpload component

```typescript
export function DocumentUpload({ applicationId }: { applicationId: string }) {
  const [uploading, setUploading] = useState(false);
  const [quota, setQuota] = useState<{ usedBytes: number; limitBytes: number } | null>(null);

  const handleDrop = async (e: React.DragEvent<HTMLDivElement>) => {
    e.preventDefault();
    const files = Array.from(e.dataTransfer.files);
    
    for (const file of files) {
      setUploading(true);
      const formData = new FormData();
      formData.append("file", file);
      formData.append("kind", "document");
      
      await fetch(`/api/documents/${applicationId}`, {
        method: "POST",
        body: formData,
      });
      setUploading(false);
    }
  };

  return (
    <div onDrop={handleDrop} className="border-2 border-dashed p-8 text-center">
      <p>Drag files here or click to upload</p>
      {quota && quota.usedBytes / quota.limitBytes > 0.8 && (
        <p className="text-yellow-600 text-sm">Approaching storage limit</p>
      )}
    </div>
  );
}
```
