### Task 1: Funnel query

```typescript
export async function getApplicationFunnel(userId: string) {
  const counts = await db.application.groupBy({
    by: ["canonicalStatus"],
    where: { ownerId: userId },
    _count: true,
  });

  return {
    applied: counts.find(c => c.canonicalStatus === "applied")?._count || 0,
    screening: counts.find(c => c.canonicalStatus === "screening")?._count || 0,
    interviewing: counts.find(c => c.canonicalStatus === "interviewing")?._count || 0,
    offer: counts.find(c => c.canonicalStatus === "offer")?._count || 0,
    rejected: counts.find(c => c.canonicalStatus === "rejected")?._count || 0,
  };
}
```

### Task 2: Time-in-stage histogram

```typescript
export async function getTimeInStageDistribution(userId: string) {
  const apps = await db.application.findMany({
    where: { ownerId: userId },
    include: { stages: { orderBy: { createdAt: "asc" } } },
  });

  const stages = apps.flatMap(app => {
    return app.stages.map(s => ({
      name: s.name,
      daysOpen: Math.ceil((s.completedAt?.getTime() || Date.now() - s.createdAt.getTime()) / (24 * 60 * 60 * 1000)),
    }));
  });

  return stages;
}
```
