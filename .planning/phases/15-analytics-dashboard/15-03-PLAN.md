```typescript
// src/app/api/analytics/export/route.ts
export async function GET(req: NextRequest) {
  const userId = await getCurrentUserId();
  const apps = await db.application.findMany({
    where: { ownerId: userId },
  });

  const csv = convertToCSV(apps);
  return new Response(csv, {
    headers: { "Content-Type": "text/csv", "Content-Disposition": "attachment; filename=foray-analytics.csv" },
  });
}
```

### Task 2: Cache analytics data

```typescript
// Cache in Redis or in-memory with TTL
const cache = new Map();
const CACHE_TTL = 5 * 60 * 1000; // 5 minutes

export async function getFunnelCached(userId: string) {
  const cacheKey = `funnel:${userId}`;
  if (cache.has(cacheKey)) return cache.get(cacheKey);
  
  const data = await getApplicationFunnel(userId);
  cache.set(cacheKey, data);
  setTimeout(() => cache.delete(cacheKey), CACHE_TTL);
  return data;
}
```
