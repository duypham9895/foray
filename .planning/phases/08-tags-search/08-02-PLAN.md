### Task 2: Create search query function

```typescript
// src/core/queries/search.ts
export async function fullTextSearch(userId: string, query: string) {
  const q = query.toLowerCase().trim();
  
  const [applications, companies, emails, stages] = await Promise.all([
    db.application.findMany({
      where: {
        ownerId: userId,
        OR: [
          { roleTitle: { search: q } },
          { notes: { search: q } },
        ],
      },
      select: { id: true, roleTitle: true, companyName: true },
      take: 10,
    }),
    db.company.findMany({
      where: {
        name: { search: q },
      },
      select: { id: true, name: true },
      take: 5,
    }),
    db.email.findMany({
      where: {
        application: { ownerId: userId },
        OR: [
          { subject: { search: q } },
          { bodyExcerpt: { search: q } },
        ],
      },
      select: { id: true, subject: true, arrivedAt: true },
      take: 10,
    }),
    db.stage.findMany({
      where: {
        application: { ownerId: userId },
        name: { search: q },
      },
      select: { id: true, name: true },
      take: 5,
    }),
  ]);

  return { applications, companies, emails, stages };
}
```

### Task 3: Create `/search` page

```typescript
// src/app/search/page.tsx
export default async function SearchPage({
  searchParams,
}: {
  searchParams: { q?: string };
}) {
  const query = searchParams.q || "";
  const user = await getCurrentUser();

  const results = await fullTextSearch(user.id, query);

  return (
    <div className="max-w-3xl mx-auto p-6">
      <div className="mb-8">
        <form method="get" action="/search" className="relative">
          <input
            type="text"
            name="q"
            defaultValue={query}
            placeholder="Search applications, companies, emails..."
            autoFocus
            className="w-full px-4 py-3 border rounded-lg text-lg"
          />
        </form>
      </div>

      {query && (
        <div className="space-y-8">
          {/* Applications */}
          {results.applications.length > 0 && (
            <section>
              <h2 className="text-lg font-semibold mb-4">Applications</h2>
              {results.applications.map(app => (
                <a key={app.id} href={`/applications/${app.id}`} className="block p-2 hover:bg-gray-100">
                  {app.roleTitle} at {app.companyName}
                </a>
              ))}
            </section>
          )}

          {/* Emails */}
          {results.emails.length > 0 && (
            <section>
              <h2 className="text-lg font-semibold mb-4">Emails</h2>
              {results.emails.map(email => (
                <div key={email.id} className="p-2 hover:bg-gray-100">
                  {email.subject}
                </div>
              ))}
            </section>
          )}

          {/* No results */}
          {results.applications.length === 0 &&
            results.emails.length === 0 &&
            results.stages.length === 0 && (
              <p className="text-gray-500">No results found for "{query}"</p>
            )}
        </div>
      )}
    </div>
  );
}
```
