### Task 2: Create tag input component with autocomplete

```typescript
// src/features/applications/components/tag-input.tsx
export function TagInput({
  value = [],
  onChange,
  allTags = [],
}: {
  value?: string[];
  onChange: (tags: string[]) => void;
  allTags?: string[];
}) {
  const [input, setInput] = useState("");
  const [filtered, setFiltered] = useState<string[]>([]);

  const handleInputChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const val = e.target.value.toLowerCase();
    setInput(val);
    setFiltered(
      allTags.filter(t => t.includes(val) && !value.includes(t))
    );
  };

  const addTag = (tag: string) => {
    const cleaned = tag.toLowerCase().trim();
    if (!value.includes(cleaned)) {
      onChange([...value, cleaned]);
    }
    setInput("");
    setFiltered([]);
  };

  return (
    <div>
      <div className="flex flex-wrap gap-2 mb-2">
        {value.map(tag => (
          <span key={tag} className="bg-blue-100 text-blue-800 px-3 py-1 rounded text-sm">
            {tag}
            <button onClick={() => onChange(value.filter(t => t !== tag))}>
              ×
            </button>
          </span>
        ))}
      </div>
      <div className="relative">
        <input
          type="text"
          value={input}
          onChange={handleInputChange}
          placeholder="Add tags (e.g., remote, startup, series-b)..."
          className="w-full border px-3 py-2 rounded"
          onKeyDown={e => {
            if (e.key === "Enter") {
              addTag(input);
            }
          }}
        />
        {filtered.length > 0 && (
          <ul className="absolute top-full left-0 right-0 bg-white border border-t-0 max-h-32 overflow-y-auto">
            {filtered.map(tag => (
              <li
                key={tag}
                onClick={() => addTag(tag)}
                className="px-3 py-2 hover:bg-gray-100 cursor-pointer"
              >
                {tag}
              </li>
            ))}
          </ul>
        )}
      </div>
    </div>
  );
}
```

### Task 3: Create filter API endpoint

```typescript
// src/app/api/applications/filter/route.ts
export async function GET(req: NextRequest) {
  const searchParams = req.nextUrl.searchParams;
  const tag = searchParams.get("tag");
  const userId = await getCurrentUserId();

  if (!tag) {
    return NextResponse.json({ error: "tag required" }, { status: 400 });
  }

  const apps = await db.application.findMany({
    where: {
      ownerId: userId,
      tags: { has: tag.toLowerCase() },
    },
    orderBy: { appliedAt: "desc" },
  });

  return NextResponse.json(apps);
}
```

### Task 4: Create tag cloud component

```typescript
// Get all tags with counts
const allTags = await db.application.findMany({
  where: { ownerId: userId },
  select: { tags: true },
});

const tagCounts = allTags
  .flatMap(app => app.tags)
  .reduce((acc, tag) => ({ ...acc, [tag]: (acc[tag] || 0) + 1 }), {});

// Render tag cloud
{Object.entries(tagCounts).map(([tag, count]) => (
  <a
    key={tag}
    href={`/applications?tag=${tag}`}
    className="text-sm text-blue-600 hover:underline"
  >
    {tag} ({count})
  </a>
))}
```
