### Task 2: Implement email auto-linking

```typescript
export async function linkEmailToRecruiter(email: Email) {
  const from = email.fromAddress;
  const recruiter = await db.recruiter.findFirst({
    where: { email: from },
  });
  
  if (recruiter) {
    await db.email.update({
      where: { id: email.id },
      data: { recruiterId: recruiter.id },
    });
  }
}
```
