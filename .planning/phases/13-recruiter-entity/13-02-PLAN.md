### Task 1: Create recruiter list page

```typescript
// src/app/recruiters/page.tsx
export default async function RecruitersPage() {
  const user = await getCurrentUser();
  const recruiters = await db.recruiter.findMany({
    where: { ownerId: user.id },
    include: { applications: true },
    orderBy: { updatedAt: "desc" },
  });

  return (
    <div className="max-w-3xl mx-auto p-6">
      <h1 className="text-2xl font-bold mb-6">Recruiters</h1>
      <div className="space-y-2">
        {recruiters.map(rec => (
          <a key={rec.id} href={`/recruiters/${rec.id}`} className="block p-4 border rounded hover:bg-gray-50">
            <div className="font-semibold">{rec.name}</div>
            <div className="text-sm text-gray-600">{rec.email}</div>
            <div className="text-sm text-gray-500">{rec.applications.length} applications</div>
          </a>
        ))}
      </div>
    </div>
  );
}
```
