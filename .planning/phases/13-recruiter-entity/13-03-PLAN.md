### Task 1: Add recruiter field to form

```typescript
export function ApplicationForm() {
  const [recruiterSearch, setRecruiterSearch] = useState("");
  const [suggestions, setSuggestions] = useState<Recruiter[]>([]);

  const handleRecruiterInput = async (val: string) => {
    setRecruiterSearch(val);
    const results = await fetch(`/api/recruiters/search?q=${val}`).then(r => r.json());
    setSuggestions(results);
  };

  return (
    <form>
      {/* ... existing fields ... */}
      <input
        type="text"
        placeholder="Recruiter name or email"
        onChange={e => handleRecruiterInput(e.target.value)}
        list="recruiter-list"
      />
      <datalist id="recruiter-list">
        {suggestions.map(rec => (
          <option key={rec.id} value={rec.email} />
        ))}
      </datalist>
    </form>
  );
}
```
