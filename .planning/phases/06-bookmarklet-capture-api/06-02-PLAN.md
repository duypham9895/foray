**Goal**: POST endpoint that receives bookmarklet data and validates

**Input**:

**Output**:

**Code**:

```typescript
// src/app/api/capture/route.ts
import { NextRequest, NextResponse } from "next/server";

const ATS_DOMAINS = [
  "greenhouse.io",
  "workday.com",
  "lever.co",
  "ashby.com",
  "smartrecruiters.com",
];

function isATSDomain(url: string): boolean {
  try {
    const hostname = new URL(url).hostname;
    return ATS_DOMAINS.some(domain => hostname.includes(domain));
  } catch {
    return false;
  }
}

export async function POST(req: NextRequest) {
  // CORS headers
  const headers = {
    "Access-Control-Allow-Origin": "*",
    "Access-Control-Allow-Methods": "POST, OPTIONS",
    "Access-Control-Allow-Headers": "Content-Type",
  };

  if (req.method === "OPTIONS") {
    return NextResponse.json({}, { status: 200, headers });
  }

  try {
    const body = await req.json();
    const { title, url, selectedText } = body;

    // Validation
    if (!url || typeof url !== "string") {
      return NextResponse.json(
        { error: "Invalid URL" },
        { status: 400, headers }
      );
    }

    if (isATSDomain(url)) {
      return NextResponse.json(
        { error: "ATS domains must be filled manually" },
        { status: 400, headers }
      );
    }

    // Create redirect with prefilled data
    const prefillData = btoa(
      JSON.stringify({
        companyName: new URL(url).hostname,
        roleTitle: title || "Unknown Role",
        selectedText: selectedText || "",
        url,
      })
    );

    return NextResponse.json(
      {
        redirectUrl: `/applications/new?prefilled=${prefillData}`,
      },
      { status: 200, headers }
    );
  } catch (error) {
    console.error("Capture API error:", error);
    return NextResponse.json(
      { error: "Internal server error" },
      { status: 500, headers }
    );
  }
}

export async function OPTIONS(req: NextRequest) {
  return POST(req);
}
```

**Verification**:
- [ ] File exists at `src/app/api/capture/route.ts`
- [ ] Endpoint responds to POST with valid JSON
- [ ] Returns 400 for ATS domains
- [ ] Returns 200 with redirectUrl for valid URLs

describe("POST /api/capture", () => {
  it("accepts valid bookmarklet POST", async () => {
    const res = await fetch("/api/capture", {
      method: "POST",
      body: JSON.stringify({
        title: "Senior Engineer - Google",
        url: "https://google.com/careers/...",
        selectedText: "preferred: 5+ years experience",
      }),
    });
    expect(res.status).toBe(200);
    const data = await res.json();
    expect(data.redirectUrl).toContain("/applications/new?prefilled=");
  });

  it("rejects ATS domains", async () => {
    const res = await fetch("/api/capture", {
      method: "POST",
      body: JSON.stringify({
        url: "https://company.greenhouse.io/jobs/123",
      }),
    });
    expect(res.status).toBe(400);
    expect(await res.json()).toHaveProperty("error");
  });

  it("handles CORS preflight", async () => {
    const res = await fetch("/api/capture", { method: "OPTIONS" });
    expect(res.headers.get("Access-Control-Allow-Origin")).toBe("*");
  });
});
