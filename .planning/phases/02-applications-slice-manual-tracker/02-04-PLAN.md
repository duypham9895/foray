    Implement helper at the top of the file:
    ```typescript
    /**
     * Toggle `target` in `currentParams.status` (comma-separated multi-select).
     * Returns the new query string (including the leading `?`) representing
     * the URL state AFTER clicking. Returns `''` when removing the last status
     * (so the URL falls back to the default filter).
     */
    function toggleStatusInUrl(currentParams: URLSearchParams, target: CanonicalStatus): string {
      const current = (currentParams.get('status') ?? '').split(',').filter(Boolean) as CanonicalStatus[]
      const next = current.includes(target)
        ? current.filter((s) => s !== target)
        : [...current, target]
      const params = new URLSearchParams(currentParams)
      if (next.length === 0) params.delete('status')
      else params.set('status', next.join(','))
      const sort = params.get('sort')
      const out = new URLSearchParams()
      if (next.length > 0) out.set('status', next.join(','))
      if (sort) out.set('sort', sort)
      const s = out.toString()
      return s ? `?${s}` : ''
    }
    ```
