### Task 1: Lever scraper

```typescript
function leverScraper(): ScraperResult {
  const roleTitle = document.querySelector(".posting-headline")?.textContent || "";
  const company = document.querySelector(".posting-company-name")?.textContent || "";
  
  return {
    roleTitle,
    company,
    location: "",
    jobDescription: document.body.innerText,
    confidence: 0.88,
  };
}
```
