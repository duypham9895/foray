## Tasks

### Task 1: Create background service worker

```typescript
// extension/background.ts
chrome.runtime.onInstalled.addListener((details) => {
  if (details.reason === "install") {
    chrome.tabs.create({ url: "https://localhost:3000/settings?extension=installed" });
  }
});

// Poll API every 5 minutes for unreviewed count
setInterval(async () => {
  const count = await fetch("http://localhost:3000/api/inbox/unreviewed-count").then(r => r.json());
  chrome.action.setBadgeText({ text: String(count) });
  chrome.action.setBadgeBackgroundColor({ color: "#FF6B6B" });
}, 5 * 60 * 1000);
```

### Task 2: Version check

```typescript
async function checkForUpdates() {
  const manifest = chrome.runtime.getManifest();
  const response = await fetch("http://localhost:3000/api/extension/version");
  const { version } = await response.json();
  
  if (version > manifest.version) {
    chrome.action.setIcon({ path: "icon-update-available.png" });
  }
}
```
