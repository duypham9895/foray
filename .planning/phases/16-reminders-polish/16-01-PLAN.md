### Task 2: Create daily reminder cron

```typescript
export async function dailyReminderCheck() {
  const reminders = await db.application.findMany({
    where: {
      followUpAt: { lte: new Date() },
      lastFollowUpAt: { lt: new Date(Date.now() - 24 * 60 * 60 * 1000) },
    },
  });

  for (const app of reminders) {
    await sendNotification(app.ownerId, {
      title: "Follow-up reminder",
      body: `Time to follow up on ${app.roleTitle} at ${app.companyName}`,
      actionUrl: `/applications/${app.id}`,
    });
  }
}
```

### Task 3: Smart suggestions

```typescript
// Suggest if >5 days since recruiter call with no follow-up
export async function suggestFollowUps(userId: string) {
  const fiveDaysAgo = new Date(Date.now() - 5 * 24 * 60 * 60 * 1000);
  return db.application.findMany({
    where: {
      ownerId: userId,
      stages: { some: { name: { contains: "recruiter" }, completedAt: { lt: fiveDaysAgo } } },
      notes: { not: { contains: "follow" } },
    },
  });
}
```
