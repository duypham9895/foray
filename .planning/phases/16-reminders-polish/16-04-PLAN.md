// src/app/api/backup/export/route.ts
export async function POST(req: NextRequest) {
  const userId = await getCurrentUserId();
  
  // Export DB: pg_dump
  const dbDump = await execSync(`pg_dump ${DATABASE_URL}`);
  
  // Package with documents
  const zip = new JSZip();
  zip.file("foray-backup.sql", dbDump);
  
  // Add documents
  const docDir = `data/uploads/${userId}`;
  for (const file of listFilesRecursive(docDir)) {
    zip.file(file, fs.readFileSync(file));
  }

  return new Response(await zip.generateAsync({ type: "blob" }), {
    headers: { "Content-Type": "application/zip" },
  });
}
