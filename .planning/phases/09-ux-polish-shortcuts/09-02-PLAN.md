### Task 2: Enhance undo toast

```typescript
// src/features/applications/components/undo-toast.tsx
export function UndoToast({ applicationId, onUndo }: UndoToastProps) {
  const [remaining, setRemaining] = useState(10);

  useEffect(() => {
    const interval = setInterval(() => setRemaining(r => r - 1), 1000);
    return () => clearInterval(interval);
  }, []);

  return (
    <div className="fixed bottom-4 right-4 bg-green-500 text-white p-4 rounded shadow-lg flex items-center gap-4">
      <span>Application auto-updated ({remaining}s)</span>
      <button onClick={onUndo} className="underline font-semibold">
        Undo
      </button>
    </div>
  );
}
```
