## Tasks

### Task 1: Create shortcut registry

```typescript
// src/features/shortcuts/shortcuts.ts
export const SHORTCUTS = {
  newApp: { key: "n", label: "New application", href: "/applications/new" },
  search: { key: "/", label: "Search", action: "focusSearch" },
  goApps: { keys: ["g", "a"], label: "Go to applications", href: "/applications" },
  goInbox: { keys: ["g", "i"], label: "Go to inbox", href: "/inbox" },
  goSettings: { keys: ["g", "s"], label: "Go to settings", href: "/settings" },
};

export function useKeyboardShortcuts() {
  const router = useRouter();
  const [lastKey, setLastKey] = useState<string | null>(null);

  useEffect(() => {
    function handleKeyDown(e: KeyboardEvent) {
      if (isInTextInput(e.target as HTMLElement)) return;

      // Single key shortcuts
      if (e.key === "n") {
        router.push(SHORTCUTS.newApp.href);
        return;
      }

      // Multi-key shortcuts (g + a, g + i, etc.)
      if (e.key === "g") {
        setLastKey("g");
        setTimeout(() => setLastKey(null), 1000);
        return;
      }

      if (lastKey === "g" && e.key === "a") {
        router.push(SHORTCUTS.goApps.href);
        setLastKey(null);
        return;
      }
      // ... etc for goInbox, goSettings
    }

    window.addEventListener("keydown", handleKeyDown);
    return () => window.removeEventListener("keydown", handleKeyDown);
  }, [lastKey]);
}
```
