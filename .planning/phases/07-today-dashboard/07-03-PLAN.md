### Task 1: Create page component and integrate all queries

```typescript
// src/app/page.tsx
import { TodaySection } from "@/features/today/components/today-section";
import { todayInterviews, actionNeeded, recent24hActivity, thisWeekCounts } from "@/core/queries/today";

export default async function TodayPage() {
  const user = await getCurrentUser();
  
  const [interviews, actionItems, activity, counts] = await Promise.all([
    todayInterviews(user.id),
    actionNeeded(user.id),
    recent24hActivity(user.id),
    thisWeekCounts(user.id),
  ]);

  return (
    <div className="max-w-2xl mx-auto p-6">
      <h1 className="text-2xl font-bold mb-8">Today</h1>

      {/* Interviews */}
      <TodaySection
        title="Interviews Today"
        icon="📅"
        isEmpty={interviews.length === 0}
        emptyMessage="No interviews scheduled"
      >
        <div className="space-y-2">
          {interviews.map(stage => (
            <InterviewCard key={stage.id} stage={stage} />
          ))}
        </div>
      </TodaySection>

      {/* Action Items */}
      <TodaySection
        title="Needs Your Attention"
        icon="⚠️"
        isEmpty={actionItems.reviewQueue.length === 0 && actionItems.staleForays.length === 0}
        emptyMessage="All caught up!"
      >
        {/* Implementation: render review queue + stale forays */}
      </TodaySection>

      {/* Recent Activity */}
      <TodaySection
        title="Last 24 Hours"
        icon="📧"
        isEmpty={activity.emails.length === 0}
        emptyMessage="No recent activity"
      >
        {/* Implementation: render emails, notes, status changes */}
      </TodaySection>

      {/* Week Summary */}
      <TodaySection
        title="This Week"
        icon="📊"
        isEmpty={false}
        emptyMessage=""
      >
        <div className="flex gap-4 flex-wrap">
          {/* CountBadges for each status */}
        </div>
      </TodaySection>
    </div>
  );
}
```

## Tests

```typescript
describe("Today page", () => {
  it("loads in <500ms with realistic data", async () => {
    const start = Date.now();
    await fetch("/");
    const elapsed = Date.now() - start;
    expect(elapsed).toBeLessThan(500);
  });

  it("displays all four sections", async () => {
    const res = await fetch("/");
    const html = await res.text();
    expect(html).toContain("Interviews Today");
    expect(html).toContain("Needs Your Attention");
    expect(html).toContain("Last 24 Hours");
    expect(html).toContain("This Week");
  });
});
```
