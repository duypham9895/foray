### Task 1: Implement `todayInterviews()` query

```typescript
// src/core/queries/today.ts
export async function todayInterviews(userId: string) {
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  
  const tomorrow = new Date(today);
  tomorrow.setDate(tomorrow.getDate() + 1);

  return db.stage.findMany({
    where: {
      application: { ownerId: userId },
      scheduledAt: {
        gte: today,
        lt: tomorrow,
      },
      cancelled: false,
    },
    include: {
      application: { select: { id: true, roleTitle: true, companyName: true } },
    },
    orderBy: { scheduledAt: "asc" },
  });
}
```

### Task 2: Implement `actionNeeded()` query

```typescript
export async function actionNeeded(userId: string) {
  // Review queue: emails waiting for triage
  const reviewQueue = await db.email.findMany({
    where: {
      application: { ownerId: userId },
      reviewRequired: true,
      resolvedAt: null,
    },
    include: {
      application: { select: { id: true, roleTitle: true } },
    },
    orderBy: { arrivedAt: "desc" },
    take: 5,
  });

  // Stale forays: no activity >7 days
  const sevenDaysAgo = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000);
  const staleForays = await db.application.findMany({
    where: {
      ownerId: userId,
      _or: [
        { updatedAt: { lt: sevenDaysAgo } },
      ],
    },
    orderBy: { updatedAt: "asc" },
    take: 5,
  });

  return { reviewQueue, staleForays };
}
```

### Task 3: Implement `recent24hActivity()` query

```typescript
export async function recent24hActivity(userId: string) {
  const since = new Date(Date.now() - 24 * 60 * 60 * 1000);

  const [emails, statusChanges, notes] = await Promise.all([
    db.email.findMany({
      where: { application: { ownerId: userId }, arrivedAt: { gte: since } },
      select: { id: true, subject: true, arrivedAt: true, applicationId: true },
      orderBy: { arrivedAt: "desc" },
      take: 5,
    }),
    db.application.findMany({
      where: { ownerId: userId, updatedAt: { gte: since } },
      select: { id: true, canonicalStatus: true, updatedAt: true },
      orderBy: { updatedAt: "desc" },
      take: 5,
    }),
    db.note.findMany({
      where: { application: { ownerId: userId }, createdAt: { gte: since } },
      select: { id: true, text: true, createdAt: true, applicationId: true },
      orderBy: { createdAt: "desc" },
      take: 5,
    }),
  ]);

  return { emails, statusChanges, notes };
}
```

### Task 4: Implement `thisWeekCounts()` query

```typescript
export async function thisWeekCounts(userId: string) {
  const now = new Date();
  const weekStart = new Date(now.setDate(now.getDate() - now.getDay()));
  const lastWeekStart = new Date(weekStart);
  lastWeekStart.setDate(lastWeekStart.getDate() - 7);

  const thisWeek = await db.application.groupBy({
    by: ["canonicalStatus"],
    where: {
      ownerId: userId,
      appliedAt: { gte: weekStart },
    },
    _count: true,
  });

  const lastWeek = await db.application.groupBy({
    by: ["canonicalStatus"],
    where: {
      ownerId: userId,
      appliedAt: { gte: lastWeekStart, lt: weekStart },
    },
    _count: true,
  });

  return { thisWeek, lastWeek };
}
```
