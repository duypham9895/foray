## Tasks

### Task 1: Add calendar sync to Stage creation

```typescript
export async function createStage(input: CreateStageInput) {
  const stage = await db.stage.create({ data: input });
  
  if (input.scheduledAt && user.hasCalendarScope) {
    const event = await createCalendarEvent(user, {
      title: `${stage.application.roleTitle} - ${stage.name}`,
      description: `Foray Stage: ${stage.name}`,
      start: { dateTime: input.scheduledAt.toISOString() },
      extendedProperties: {
        private: { forayStageId: stage.id },
      },
    });
    
    await db.stage.update({
      where: { id: stage.id },
      data: { googleCalendarEventId: event.id },
    });
  }
  
  return stage;
}
```
