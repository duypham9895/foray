## Tasks

### Task 1: Create cron job to sync Calendar → Foray

```typescript
// src/jobs/syncCalendarEvents.ts
export async function syncCalendarEvents(userId: string) {
  const user = await db.user.findUnique({ where: { id: userId } });
  if (!user?.googleCalendarToken || !user.hasCalendarScope) return;

  const events = await googleCalendar.events.list({
    calendarId: "primary",
    showDeleted: true,
    updatedMin: new Date(Date.now() - 24 * 60 * 60 * 1000).toISOString(),
  });

  for (const event of events.items) {
    const stageId = event.extendedProperties?.private?.forayStageId;
    if (!stageId) continue;

    const stage = await db.stage.findUnique({ where: { id: stageId } });
    if (!stage) continue;

    // Foray is source of truth: only update Stage if Calendar event was deleted
    if (event.status === "cancelled") {
      await db.stage.update({
        where: { id: stageId },
        data: { cancelled: true },
      });
    }
  }
}
```
